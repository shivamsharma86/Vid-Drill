            trending: 'üî•',
            roast: 'üçó',
            cricket: 'üèè'
        };
        
        let currentFilter = 'all';
        let currentSearchQuery = '';
        let itemsPerPage = 6;
        let currentPage = 1;
        
        // User data storage
        let userData = {
            favorites: JSON.parse(localStorage.getItem('vidDrillFavorites')) || [],
            watchLater: JSON.parse(localStorage.getItem('vidDrillWatchLater')) || [],
            recentlyWatched: JSON.parse(localStorage.getItem('vidDrillRecentlyWatched')) || []
        };
        
        // Save user data to localStorage
        const saveUserData = () => {
            localStorage.setItem('vidDrillFavorites', JSON.stringify(userData.favorites));
            localStorage.setItem('vidDrillWatchLater', JSON.stringify(userData.watchLater));
            localStorage.setItem('vidDrillRecentlyWatched', JSON.stringify(userData.recentlyWatched));
        };
        
        // Show toast notification
        const showToast = (message, type = 'success') => {
            toastMessage.textContent = message;
            toast.className = `toast ${type}`;
            toastIcon.textContent = type === 'success' ? '‚úÖ' : '‚ö†Ô∏è';
            
            toast.classList.add('show');
            
            setTimeout(() => {
                toast.classList.remove('show');
            }, 3000);
        };
        
        // Setup notifications
        const setupNotifications = () => {
            const newVideos = videoData.filter(v => v.isNew);
            if (newVideos.length > 0) {
                notificationCount.innerText = newVideos.length;

                let panelHTML = '<h4>New Videos <button class="clear-notifications">Clear All</button></h4>';
                newVideos.forEach(video => {
                    panelHTML += `
                        <div class="notification-item" data-video-id="${video.id}">
                            <img src="${video.thumbnail}" alt="${video.title}">
                            <div>
                                <p>${video.title}</p>
                                <span>${video.category} ‚Ä¢ ${new Date(video.uploadDate).toLocaleDateString()}</span>
                            </div>
                        </div>
                    `;
                });
                notificationPanel.innerHTML = panelHTML;
                
                // Add event listener for clear notifications
                document.querySelector('.clear-notifications').addEventListener('click', (e) => {
                    e.stopPropagation();
                    notificationPanel.style.display = 'none';
                    notificationCount.style.display = 'none';
                    showToast('All notifications cleared', 'success');
                });
            }
        };
        
        // Setup FAQ accordion
        const setupFAQ = () => {
            faqItems.forEach(item => {
                const question = item.querySelector('.faq-question');
                question.addEventListener('click', () => {
                    // Close all other items
                    faqItems.forEach(otherItem => {
                        if (otherItem !== item) {
                            otherItem.classList.remove('active');
                        }
                    });
                    
                    // Toggle current item
                    item.classList.toggle('active');
                });
            });
        };
        
        // Create video card
        const createVideoCard = (video) => {
            const isFavorite = userData.favorites.includes(video.id);
            const isWatchLater = userData.watchLater.includes(video.id);
            
            const card = document.createElement('div');
            card.className = `video-card ${video.isNew ? 'new' : ''}`;
            card.dataset.videoId = video.id;
            card.innerHTML = `
                <div class="thumbnail-container">
                    <img src="${video.thumbnail}" alt="${video.title}" class="thumbnail">
                    <span class="category-tag">${categoryEmojis[video.category]} ${video.category.charAt(0).toUpperCase() + video.category.slice(1)}</span>
                </div>
                <div class="card-content">
                    <h3 class="video-title">${video.title}</h3>
                    <p class="video-description">${video.description}</p>
                    <div class="video-meta">
                        <span><i class="far fa-calendar"></i> ${new Date(video.uploadDate).toLocaleDateString()}</span>
                    </div>
                    <div class="button-group">
                        <button class="btn btn-watch"><i class="fas fa-play"></i> Watch</button>
                        <a href="${video.downloadLink || '#'}" class="btn btn-download" download><i class="fas fa-download"></i> Download</a>
                        <button class="action-btn btn-share" title="Share"><i class="fas fa-share"></i></button>
                        <button class="action-btn btn-watch-later ${isWatchLater ? 'active' : ''}" title="Watch Later"><i class="${isWatchLater ? 'fas' : 'far'} fa-bookmark"></i></button>
                        <button class="action-btn btn-favorite ${isFavorite ? 'active' : ''}" title="Add to Favorites"><i class="${isFavorite ? 'fas' : 'far'} fa-heart"></i></button>
                    </div>
                </div>
            `;
            return card;
        };
        
        // Render videos
        const renderVideos = (videoList = videoData) => {
            let filteredVideos = [...videoList];
            
            if (currentFilter !== 'all') {
                filteredVideos = filteredVideos.filter(v => v.category === currentFilter);
            }
            
            if (currentSearchQuery) {
                filteredVideos = filteredVideos.filter(v => 
                    v.title.toLowerCase().includes(currentSearchQuery) || 
                    v.description.toLowerCase().includes(currentSearchQuery)
                );
            }
            
            const videosToShow = filteredVideos.slice(0, currentPage * itemsPerPage);
            videoGrid.innerHTML = '';
            
            videosToShow.forEach(video => {
                videoGrid.appendChild(createVideoCard(video));
            });
            
            loadMoreBtn.style.display = videosToShow.length >= filteredVideos.length ? 'none' : 'block';
        };
        
        // Show special list (Favorites, Watch Later, Recently Watched)
        const showSpecialList = (listType) => {
            let videoIds = [];
            let listName = '';
            
            switch(listType) {
                case 'favorites':
                    videoIds = userData.favorites;
                    listName = 'Favorites';
                    break;
                case 'watchLater':
                    videoIds = userData.watchLater;
                    listName = 'Watch Later';
                    break;
                case 'recentlyWatched':
                    videoIds = userData.recentlyWatched;
                    listName = 'Recently Watched';
                    break;
            }
            
            const videos = videoIds.map(id => videoData.find(v => v.id === id)).filter(Boolean);
            
            if (videos.length === 0) {
                showToast(`No videos in your ${listName} list`, 'warning');
                return;
            }
            
            currentFilter = 'special';
            currentSearchQuery = ''; 
            searchBar.value = '';
            currentPage = 1;
            
            // Clear grid and show special videos
            videoGrid.innerHTML = '';
            videos.forEach(video => {
                videoGrid.appendChild(createVideoCard(video));
            });
            
            loadMoreBtn.style.display = 'none';
            sidenav.style.width = '0';
            showToast(`Showing your ${listName}`, 'success');
        };
        
        // Event listeners
        navLinks.forEach(link => {
            link.addEventListener('click', () => {
                navLinks.forEach(l => l.classList.remove('active'));
                link.classList.add('active');
                currentFilter = link.dataset.filter;
                currentPage = 1;
                renderVideos();
            });
        });
        
        categoryCards.forEach(card => {
            card.addEventListener('click', () => {
                categoryCards.forEach(c => c.classList.remove('active'));
                card.classList.add('active');
                currentFilter = card.dataset.category;
                currentPage = 1;
                renderVideos();
            });
        });
        
        searchBar.addEventListener('input', () => {
            currentSearchQuery = searchBar.value.toLowerCase();
            currentPage = 1;
            renderVideos();
        });
        
        loadMoreBtn.addEventListener('click', () => {
            currentPage++;
            renderVideos();
        });
        
        openNavBtn.addEventListener('click', () => sidenav.style.width = '250px');
        closeNavBtn.addEventListener('click', () => sidenav.style.width = '0');
        
        // Sidebar navigation
        homeBtn.addEventListener('click', (e) => {
            e.preventDefault();
            currentFilter = 'all';
            currentSearchQuery = ''; 
            searchBar.value = '';
            currentPage = 1;
            renderVideos();
            sidenav.style.width = '0';
            document.querySelector('.nav-link.active').classList.remove('active');
            document.querySelector('.nav-link[data-filter="all"]').classList.add('active');
            categoryCards.forEach(c => c.classList.remove('active'));
            document.querySelector('.category-card[data-category="all"]').classList.add('active');
            showToast('Back to home', 'success');
        });
        
        favoritesBtn.addEventListener('click', (e) => {
            e.preventDefault();
            showSpecialList('favorites');
        });
        
        watchLaterBtn.addEventListener('click', (e) => {
            e.preventDefault();
            showSpecialList('watchLater');
        });
        
        recentlyWatchedBtn.addEventListener('click', (e) => {
            e.preventDefault();
            showSpecialList('recentlyWatched');
        });
        
        notificationBell.addEventListener('click', (e) => {
            e.stopPropagation();
            notificationPanel.style.display = notificationPanel.style.display === 'block' ? 'none' : 'block';
        });
        
        document.addEventListener('click', (e) => {
            if (!e.target.closest('.notification-container')) {
                notificationPanel.style.display = 'none';
            }
        });
        
        // Video card interactions
        videoGrid.addEventListener('click', (e) => {
            const card = e.target.closest('.video-card');
            if (!card) return;
            
            const videoId = parseInt(card.dataset.videoId);
            const video = videoData.find(v => v.id === videoId);

            if (e.target.matches('.btn-watch') || e.target.closest('.btn-watch')) {
                if (video) {
                    // Add to recently watched
                    userData.recentlyWatched = userData.recentlyWatched.filter(id => id !== video.id);
                    userData.recentlyWatched.unshift(video.id);
                    if (userData.recentlyWatched.length > 20) userData.recentlyWatched.pop();
                    saveUserData();
                    
                    showToast(`Now playing: ${video.title}`, 'success');
                }
            }

            if (e.target.matches('.btn-download') || e.target.closest('.btn-download')) {
                showToast(`Download started for: ${video.title}`, 'success');
            }

            if (e.target.matches('.btn-share') || e.target.closest('.btn-share')) {
                showToast(`Link copied to clipboard for sharing`, 'success');
            }

            if (e.target.matches('.btn-watch-later') || e.target.closest('.btn-watch-later')) {
                const button = e.target.closest('.btn-watch-later');
                if (userData.watchLater.includes(videoId)) {
                    userData.watchLater = userData.watchLater.filter(id => id !== videoId);
                    button.classList.remove('active');
                    button.innerHTML = '<i class="far fa-bookmark"></i>';
                    showToast('Removed from Watch Later', 'success');
                } else {
                    userData.watchLater.push(videoId);
                    button.classList.add('active');
                    button.innerHTML = '<i class="fas fa-bookmark"></i>';
                    showToast('Added to Watch Later', 'success');
                }
                saveUserData();
            }
            
            if (e.target.matches('.btn-favorite') || e.target.closest('.btn-favorite')) {
                const button = e.target.closest('.btn-favorite');
                if (userData.favorites.includes(videoId)) {
                    userData.favorites = userData.favorites.filter(id => id !== videoId);
                    button.classList.remove('active');
                    button.innerHTML = '<i class="far fa-heart"></i>';
                    showToast('Removed from Favorites', 'success');
                } else {
                    userData.favorites.push(videoId);
                    button.classList.add('active');
                    button.innerHTML = '<i class="fas fa-heart"></i>';
                    showToast('Added to Favorites', 'success');
                }
                saveUserData();
            }
        });
        
        // Footer link handlers
        footerLinks.forEach(link => {
            link.addEventListener('click', (e) => {
                e.preventDefault();
                const page = link.getAttribute('data-page');
                showToast(`Opening ${link.textContent} page`, 'success');
                
                // You can add actual navigation logic here
                // For now, we'll just show the toast
            });
        });
        
        // Initialize
        setupNotifications();
        setupFAQ();
        renderVideos();
    });
    </script>
</body>
